46 /** \brief the types of data in one JKQTdatastoreItem
47  * \ingroup jkqtpdatastorage
48  *
49  * \c JKQTPSingleColumn:
50  * \verbatim
51 +-----+-----+-----+....................+-----+
52 | 0 | 1 | 2 | + N-1 |
53 +-----+-----+-----+....................+-----+
54 \endverbatim
55  *
56  *
57  * \c JKQTPMatrixColumn:
58  * \verbatim
59  ================= COLUMN 1 ================= ================= COLUMN 2 =================
60 +-----+-----+-----+....................+-----++-----+-----+-----+....................+-----+ .....
61 | 0 | 1 | 2 | + N-1 || 0 | 1 | 2 | + N-1 |
62 +-----+-----+-----+....................+-----++-----+-----+-----+....................+-----+ .....
63  =R1== =R2== =R3== =RN== =R1== =R2== =R3== =RN==
64 \endverbatim
65  *
66  *
67  * \c JKQTPMatrixRow (C standard representation of matrices):
68  * \verbatim
69  ================== ROW 1 ================== ================== ROW 2 ==================
70 +-----+-----+-----+....................+-----++-----+-----+-----+....................+-----+ .....
71 | 0 | 1 | 2 | + N-1 || 0 | 1 | 2 | + N-1 |
72 +-----+-----+-----+....................+-----++-----+-----+-----+....................+-----+ .....
73  =C1== =C2== =C3== =CN== =C1== =C2== =C3== =CN==
74 \endverbatim
75  */

82 /** \brief This class manages chunks of memory that are used for column data in JKQTBasePlotter descendents
83  * \ingroup jkqtpdatastorage
84  *
85  * This class manages a list if JKQTPDatastoreItem onjects that may each contain a chunk of memory, containig
86  * one or more columns of data. Each item can be accessed with get() by a specific ID which is returned by add().
87  * JKQTPColumn. You may only clear all chunks of memory/items. If you no longer need some of the data, but still want
88  * to access the rest you will simply have to destroy all JKQTPColumn that point to the item with their
89  * JKQTPColumns:datastoreItem property.
90  *
91  * \verbatim
92

97 | | 0 # # # # # # # # # # #<--|---|-------------------|---datastoreOffset = 0 |
98 | | | | | |
99 | +- JKQTPDatastoreItem --------+ | +- JKQTPColumn ----------------+
100 | 1 | JKQTPSingleColumn |<--|---\ 1 | datastore |
101 | | 0 # # # # # # # # # # #<--|---|-\ \--------------|---datastoreItem = 1 |
102 | | | | \----------------|---datastoreOffset = 0 |
103 | +- JKQTPDatastoreItem --------+ | | |
104 | : : | +------------------------------+
105 | : : | : :
106 | : : | : :
107 | +- JKQTPDatastoreItem --------+ | : :
108 |N-1 | JKQTPMatrixRow |<--|---\ : :
109 | | 0 # # # # # # # # # # # | | \ +- JKQTPColumn ----------------+
110 | | 1 # # # # # # # # # # # | | \ M-1 | datastore |
111 | | 2 # # # # # # # # # # #<--|---|--\ \------------|---datastoreItem = N-1 |

118 \endverbatim

266  /** \brief add one external column to the datastore. It will be filled with the contents of vector \a data.
267  *
268  * \tparam TContainer datatype of the container, which need to support standard C++ iterators and the function \c size(). The contents needs to be convertible to double.
269  * \param data data vector to be copied
270  * \param name name for the column
271  * \param stride strides through the container \a data with the given stride
272  * \param start starts copying from \a data with the element \a start
273  * \return the ID of the newly created column
274  */
275  template <typename TContainer>
276  size_t addCopiedColumn(const TContainer& data, const QString& name, size_t stride, size_t start=0) {
277  const size_t N=static_cast<size_t>(data.size()-start)/stride;
278  double* d=static_cast<double*>(malloc(static_cast<size_t>(N)*sizeof(double)));
279  if (N>0) {
280  size_t r=0;
281  auto it=data.begin();
282  if (start>0) it+=start;
283  for (; it!=data.end(); it+=stride) {
284  d[r]=jkqtp_todouble(*it);
285  r++;
286  }
287  }
288  size_t itemid=addInternalItem(d, N);
289  return addColumnForItem(itemid, 0, name);

720 /** \brief this represents one chunk of memory which can represent one or more columns of data for JKQTBasePlotter.
721  * See JKQTPDatastore for more information.

724  * Each chunk of memory is pointed at by a simple double* pointer \c data. the memory layout of the memory layout of
725  * the RAM segment pointed at by \c data is determined by the parameter \c dataformat:

729  * getColumns() and getRows() ). This class may manage chunks of "internal" and "external" memory (which is indicated by
730  * the boolean property \c internal. Internal memory will be allocated (in the constructor) and freed (in the destructor) by this
731  * object. External memory may be accessed via this class, but will neither by allocated nor freed. These tasks are up to the
732  * user. Use this option, if you don't want to generate copies of huge datasets, or you want to change the data while the program
733  * is running and don't want to do the change at more than one spot.
