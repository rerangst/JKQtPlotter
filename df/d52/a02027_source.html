49  /** \brief internal helper function for JKQTPCopyCvMatToColumn()
50  * \ingroup jkqtpopencvinterface
51  * \internal */
52  template <typename TPixel>
53  inline void copyDataFromMat(double* data, const cv::Mat& mat, int channel) {
54  size_t r=0;
55  const int channels=mat.channels();
56  for (int iy=0; iy<mat.rows; iy++ ) {
57  const TPixel* row=mat.ptr<TPixel>(iy);
58  for (int ix=0; ix<mat.cols; ix++ ) {
59  data[r]=jkqtp_todouble(*static_cast<const TPixel*>(&(row[ix*channels+channel])));
60  r++;
61  }
62  }
63  }
64 }

66 inline size_t JKQTPCopyCvMatToColumn(JKQTPDatastore* datastore, const cv::Mat& mat, const QString &name, int channel)
67 {
68  const size_t N=static_cast<size_t>(mat.cols*mat.rows);
69  double* d=static_cast<double*>(malloc(static_cast<size_t>(N)*sizeof(double)));
70 
71  if (CV_MAT_DEPTH(mat.type())==CV_64F) JKQTPDatastore_Helper::copyDataFromMat<double>(d, mat, channel);
72  else if (CV_MAT_DEPTH(mat.type())==CV_32F) JKQTPDatastore_Helper::copyDataFromMat<float>(d, mat, channel);
73  else if (CV_MAT_DEPTH(mat.type())==CV_32S) JKQTPDatastore_Helper::copyDataFromMat<uint32_t>(d, mat, channel);
74  else if (CV_MAT_DEPTH(mat.type())==CV_16S) JKQTPDatastore_Helper::copyDataFromMat<int16_t>(d, mat, channel);
75  else if (CV_MAT_DEPTH(mat.type())==CV_16U) JKQTPDatastore_Helper::copyDataFromMat<uint16_t>(d, mat, channel);
76  else if (CV_MAT_DEPTH(mat.type())==CV_8S) JKQTPDatastore_Helper::copyDataFromMat<int8_t>(d, mat, channel);
77  else if (CV_MAT_DEPTH(mat.type())==CV_8U) JKQTPDatastore_Helper::copyDataFromMat<uint8_t>(d, mat, channel);
78  else throw std::runtime_error("datatype of cv::Mat not supported by JKQTPDatastore::copyImageToColumn()");
79 
80  size_t itemid=datastore->addInternalItem(d, N);
81  return datastore->addColumnForItem(itemid, 0, name);
82 }
