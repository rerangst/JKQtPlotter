35 /*! \brief This implements vertical <a href="http://en.wikipedia.org/wiki/Box_plot">boxplots</a>
36  \ingroup jkqtplotter_statgraphs
37 
38  The x position is given in posColumn. All other data are given in the medianColumn, minColumn, maxColumn,
39  percentile25Column and percentile75Column.
40 
41  \image html plot_boxplotvertical.png
42 
43  The different features of a boxplot are:
44 
45  \image html boxplots.png
46 
47  \note There are additional classes to draw a single boxplot element: JKQTPBoxplotHorizontalElement and JKQTPBoxplotVerticalElement.
48  In these you can set the data values, as they are NOT drawn from a data column. This can be useful, if you e.g. want to
49  draw the statistical properties of a distribution.
50 
51  \section BoxplotOutliers Outliers
52 
53  If you also want to display outliers, as shown here with circles:
54 
55  \image html boxplot_outliers.png
56 
57  You need to add them as (x,y)-pairs to the datastore and add a separate JKQTPXYLineGraph that shows these. See \ref JKQTPlotterBoxplotsGraphs for details. Here is an example code-snippet:
58 
59  \code
60  // 4. create a graph of vertical boxplots:
61  JKQTPBoxplotVerticalGraph* graph=new JKQTPBoxplotVerticalGraph(&plot);
62  graph->setPosColumn(columnPOS);
63  graph->setMinColumn(columnMIN);
64  graph->setPercentile25Column(columnQ25);
65  graph->setMedianColumn(columnMEDIAN);
66  graph->setMeanColumn(columnMEAN);
67  graph->setPercentile75Column(columnQ75);
68  graph->setMaxColumn(columnMAX);
69  graph->setTitle("vertical Boxplots");
70 
71  // 5. outliers need to be drawn separately
72  JKQTPXYLineGraph* graphOutliers=new JKQTPXYLineGraph(&plot);
73  graphOutliers->setXColumn(columnOUTLIERSX);
74  graphOutliers->setYColumn(columnOUTLIERSY);
75  graphOutliers->setTitle("outliers");
76  // make the color a darker shade of the color of graph
77  graphOutliers->setColor(graph->getColor().darker());
78  graphOutliers->setFillColor(QColor("white"));
79  // draw outliers as small circles, without lines
80  graphOutliers->setSymbol(JKQTPCircle);
81  graphOutliers->setDrawLine(false);
82  graphOutliers->setSymbolSize(7);
83  \endcode
84 
85  \see \ref JKQTPlotterBoxplotsGraphs
86 
87  */
