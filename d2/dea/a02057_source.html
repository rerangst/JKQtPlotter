226 {
227  if (!dbl || width<=0 || height<=0)
228  return 0;
229 
230  double min = 0;
231  double max = 0;
232  bool first=true;
233  for (int i=1; i<width*height; ++i)
234  {
235  T v=dbl[i];
236  if (!(std::isnan(static_cast<long double>(v)) || std::isinf(static_cast<long double>(v)))) {
237  if (first) {
238  min=max=v;
239  first=false;
240  } else {
241  if (v < min)
242  min = v;
243  else if (v > max)
244  max = v;
245  }
246  }
247  }
248  return min;

309  {

311  if (std::isfinite(static_cast<long double>(v))) {
312  if (first) {
313  min=max=v;
314  first=false;

615 /*! \brief convert a 2D image (as 1D array) into a QImage with given palette (see JKQTFPColorPalette)

617 
618  This method uses lookup tables which are saved as static variables to convert the 2D array into
619  an image. The luts are only created once, and stored then, so mor CPU time is saved. The precompiler define
620  JKQTPImageTools::LUTSIZE sets the size of the LUTs. Note that if you don't use a specific color palette,
621  the according LUT won't be calculated and stored!
622 
623  \param dbl_in pointer to a 1D array of template type \c T representing the image to plot. This array has to be of size \a width * \a height
624  \param width width of the array in \a dbl
625  \param height height of the array in \a dbl
626  \param[out] im the QImage object to draw to (should be initialized as \c QImage::Format_ARGB32 )
627  \param palette the color palette to use for the display
628  \param minColor lower boundary of color range in \a dbl pixels, if \a minColor == \a maxColor then this function will extract the image min and image max.
629  \param maxColor upper boundary of color range in \a dbl pixels, if \a minColor == \a maxColor then this function will extract the image min and image max.
630  \param paletteMinFail specifies what shell happen, when a value in \a dbl is below \a minColor
631  \param paletteMaxFail specifies what shell happen, when a value in \a dbl is above \a maxColor
632  \param minFailColor color to use for pixels that are below \a minColor for some settings of \a paletteMinFail
633  \param maxFailColor color to use for pixels that are below \a maxColor for some settings of \a paletteMaxFail
634  \param nanColor color to use for pixels that are not-a-number
635  \param infColor color to use for pixels that are infinity
636 */
637 template <class T>
638 inline void JKQTPImagePlot_array2image(const T* dbl_in, int width, int height, QImage &img, JKQTPMathImageColorPalette palette, double minColor, double maxColor, JKQTPMathImageColorRangeFailAction paletteMinFail=JKQTPMathImageLastPaletteColor, JKQTPMathImageColorRangeFailAction paletteMaxFail=JKQTPMathImageLastPaletteColor, QColor minFailColor=QColor("black"), QColor maxFailColor=QColor("black"), QColor nanColor=QColor("black"), QColor infColor=QColor("black"), bool logScale=false, double logBase=10.0, const int* lutUser=0, int lutUserSize=0)
639 {
640  if (!dbl_in || width<=0 || height<=0)
641  return;
642 
643  double min = *dbl_in;
644  double max = *dbl_in;
645  if (minColor == maxColor) {
646  bool first=true;
647  for (int i=1; i<width*height; ++i)
648  {
649  T v=dbl_in[i];
650  if (!(std::isnan(static_cast<long double>(v)) || std::isinf(static_cast<long double>(v)))) {
651  if (first) {
652  min=max=v;
653  first=false;
654  } else {
655  if (v < min)
656  min = v;
657  else if (v > max)
658  max = v;
659  }
660  }
661  }
662 
663  } else {
664  min = minColor;
665  max = maxColor;
666  }
667 
668  const T* dbl=dbl_in;
669  T* dbl1=nullptr;
670  if (logScale) {
671  double logB=log10(logBase);
672  dbl1=(T*)malloc(width*height*sizeof(T));
673  //memcpy(dbl, dbl_in, width*height*sizeof(T));
674  for (int i=0; i<width*height; i++) {
675  dbl1[i]=log10(dbl_in[i])/logB;
676  }
677  dbl=dbl1;
678  min=log10(min)/logB;
679  max=log10(max)/logB;
680  }
681  double delta=max-min;
682 
683 
684  const int* lut_used=nullptr;
685  int lutSize=JKQTPImageTools::LUTSIZE;
686  if (global_jkqtpimagetools_lutstore.size()<=0) global_jkqtpimagetools_lutstore=JKQTPImagePlot_makeQList<int*>(nullptr, JKQTPImagePlot_getPredefinedPalettes().size()+2);
687 
688 
689  img = QImage(width, height, QImage::Format_ARGB32);
690  if (min == max)
691  img.fill(0);
692  else
693  {
694 
695  if (palette==JKQTPMathImageUSER_PALETTE) {
696  lut_used=lutUser;
697  lutSize=lutUserSize;
698  //qDebug()<<"user palette "<<lutUser<<lutUserSize;
699  } else {
700  lut_used=JKQTPImagePlot_getCreateLUT(global_jkqtpimagetools_lutstore, palette);
701  }
702 
703 
704  if (lut_used!=nullptr && lutSize>0) {
705  const unsigned int* lut_usedui=reinterpret_cast<const unsigned int*>(lut_used);
706  // LUT found: collor the image accordingly
707  for (int j=0; j<height; ++j) {
708  QRgb* line=reinterpret_cast<QRgb*>(img.scanLine(height-1-j));
709  for (int i=0; i<width; ++i) {
710  double val=dbl[j*width+i];
711  if (std::isnan(val)) {
712  line[i]=nanColor.rgba();
713  } else if (std::isinf(val)) {
714  line[i]=infColor.rgba();
715  } else {
716  const int v = static_cast<int>((val-min)/delta*static_cast<double>(lutSize));
717  const int vv = (v < 0) ? 0 : ( (v > lutSize) ? (lutSize) : v);
718  line[i]=lut_usedui[vv];
719  if ((v<0)&&(paletteMinFail==JKQTPMathImageGivenColor)) {
720  line[i]=minFailColor.rgba();
721  } else if ((v>lutSize)&&(paletteMaxFail==JKQTPMathImageGivenColor)) {
722  line[i]=maxFailColor.rgba();
723  } else if ((v<0)&&(paletteMinFail==JKQTPMathImageTransparent)) {
724  line[i]=QColor(Qt::transparent).rgba();
725  } else if ((v>lutSize)&&(paletteMaxFail==JKQTPMathImageTransparent)) {
726  line[i]=QColor(Qt::transparent).rgba();
727  }
728  }
729  }
730  }
731  } else {
732  // no LUT found: paint a black image!
733  img.fill(0);
734  }
735  }
736 
737  if (dbl1) free(dbl1);
738 
739 };
