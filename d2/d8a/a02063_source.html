91  JKQTPFinalAct(JKQTPFinalAct&& other) noexcept

99  JKQTPFinalAct& operator=(const JKQTPFinalAct&) = delete;
100

102  {

104  }
105 
106 private:

109 };

618  \ingroup jkqtptools
619 
620  implementation from http://www.linux-related.de/index.html?/coding/sort/sort_quick.htm
621 */
622 template <class T>
623 inline void jkqtpQuicksort(T* a, int l, int r){
624  if(r>l){
625  int i=l-1;
626  int j=r;
627 
628  for(;;){
629  while(a[++i]<a[r]);
630  while(a[--j]>a[r] && j>i);
631  if(i>=j) break;
632  jkqtpSwap<T>(a, i, j);
633  }
634  jkqtpSwap<T>(a, i, r);
635 
636  jkqtpQuicksort(a, l, i-1);
637  jkqtpQuicksort(a, i+1, r);
638  }
639 }

671 /*! \brief sort the given arrays, using \a input as sort criterion
672  \ingroup jkqtptools
673 
674  \param input array to be sorted
675  \param input2 array to be sorted
676  \param N size of the array input
677  \param output if \c !=nullptr data is written here (the memory location pointed at by \a output has to have at least the length \a N !!!),
678  otherwise the array input is sorted inplace.
679  \param output2 if \c !=nullptr data is written here (the memory location pointed at by \a output has to have at least the length \a N !!!),
680  otherwise the array input is sorted inplace.
681 
682  */
683 template <class T, class T2>
684 inline void jkqtpSort(T* input, T2* input2, int N, T* output=nullptr, T2* output2=nullptr) {
685  if ((!input)) return ;
686  if (N<=0) return;
687  T* data=input;
688  if (output!=nullptr) {
689  data=output;
690  memcpy(output, input, N*sizeof(T));
691  }
692  T2* data2=input2;
693  if (output2!=nullptr && input2!=nullptr) {
694  data2=output2;
695  memcpy(output2, input2, N*sizeof(T2));
696  }
697  jkqtpQuicksort(data, data2, 0, N-1);
698 }
